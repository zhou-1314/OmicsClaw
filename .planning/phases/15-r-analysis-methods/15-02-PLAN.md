Permutation logic (from RunProportionTest.R / .permutation_test):
```r
# For a comparison group1 vs group2:
# 1. Filter cells belonging to group1 or group2 (split.by values)
# 2. Count cells per cluster per condition → observed proportions
# 3. Compute observed log2(prop_group1 / prop_group2) per cluster (obs_log2FD)
# 4. Permute cell condition labels n_permutations times:
#    a. Randomly shuffle condition labels among cells
#    b. Compute permuted log2FD per cluster
#    c. Store all permuted log2FDs
# 5. pval = fraction of permuted |log2FD| >= |obs_log2FD|
# 6. FDR = p.adjust(pval, method="BH")
# 7. boot_CI_2.5, boot_CI_97.5 = quantile(permuted_log2FDs, c(0.025, 0.975))
# 8. boot_mean_log2FD = mean(permuted_log2FDs)
```

```r
.run_permutation <- function(meta, group_col, split_col, group1, group2, n_perm) {
  # Filter cells in this comparison
  sub <- meta[meta[[split_col]] %in% c(group1, group2), ]
  if (nrow(sub) < 10) return(NULL)

  clusters <- sort(unique(as.character(sub[[group_col]])))

  # Observed proportions
  count_observed <- table(sub[[split_col]], sub[[group_col]])
  # Normalise per condition to proportions
  prop_obs <- sweep(count_observed, 1, rowSums(count_observed), "/")
  # guard against zero-proportion (add pseudocount)
  p1 <- prop_obs[group1, ] + 1e-9
  p2 <- prop_obs[group2, ] + 1e-9
  obs_log2FD <- log2(p1 / p2)

  # Permutation
  labels <- sub[[split_col]]
  n <- length(labels)
  n1 <- sum(labels == group1)
  boot_log2FD <- matrix(NA_real_, nrow=n_perm, ncol=length(clusters),
                        dimnames=list(NULL, clusters))
  set.seed(42)
  for (i in seq_len(n_perm)) {
    perm_labels <- sample(labels, n, replace=FALSE)
    # Count per cluster per permuted condition
    ct <- table(perm_labels, sub[[group_col]])
    if (!all(c(group1, group2) %in% rownames(ct))) next
    pp1 <- ct[group1, ] / (sum(ct[group1, ]) + 1e-9) + 1e-9
    pp2 <- ct[group2, ] / (sum(ct[group2, ]) + 1e-9) + 1e-9
    boot_log2FD[i, ] <- log2(pp1 / pp2)
  }

  pval <- sapply(clusters, function(cl) {
    mean(abs(boot_log2FD[, cl]) >= abs(obs_log2FD[cl]), na.rm=TRUE)
  })
  fdr <- p.adjust(pval, method="BH")
  boot_mean <- colMeans(boot_log2FD, na.rm=TRUE)
  boot_ci_lo <- apply(boot_log2FD, 2, quantile, probs=0.025, na.rm=TRUE)
  boot_ci_hi <- apply(boot_log2FD, 2, quantile, probs=0.975, na.rm=TRUE)

  data.frame(
    clusters      = clusters,
    group1        = group1,
    group2        = group2,
    comparison    = paste0(group1, "_vs_", group2),
    obs_log2FD    = as.numeric(obs_log2FD[clusters]),
    pval          = pval,
    FDR           = fdr,
    boot_mean_log2FD = boot_mean,
    boot_CI_2.5   = boot_ci_lo,
    boot_CI_97.5  = boot_ci_hi,
    stringsAsFactors = FALSE
  )
}
```

```python
def _run_proportion_test_r(adata, condition_key, cell_type_key, output_dir, params):
    import warnings
    r_script = R_SCRIPTS_DIR / "sc_proportion_test_r.R"
    if not r_script.exists():
        raise FileNotFoundError(f"R script not found: {r_script}")

    r_work_dir = output_dir / "r_work"
    r_work_dir.mkdir(parents=True, exist_ok=True)

    # Export cell metadata for R (cell_id + relevant columns only)
    meta_cols = [cell_type_key, condition_key]
    meta_df = adata.obs[meta_cols].copy().reset_index()
    meta_df.columns = ["cell_id"] + meta_cols
    meta_csv = r_work_dir / "cell_meta.csv"
    meta_df.to_csv(meta_csv, index=False)

    comparison = params.get("contrast", "auto")
    n_perm = str(params.get("n_permutations", 1000))

    runner = RScriptRunner(timeout=300)
    result_csv = r_work_dir / "proportion_test_results.csv"
    r_success = True
    try:
        runner.run_script(
            r_script,
            args=[str(meta_csv), str(r_work_dir), cell_type_key,
                  condition_key, comparison, n_perm],
            expected_outputs=["proportion_test_results.csv"],
            output_dir=r_work_dir,
        )
    except RScriptError as exc:
        warnings.warn(f"proportion_test_r R script failed (Python figures unaffected): {exc}")
        r_success = False

    result_df = pd.DataFrame()
    if r_success and result_csv.exists():
        try:
            result_df = pd.read_csv(result_csv)
        except Exception as exc:
            warnings.warn(f"Could not read proportion_test_r results: {exc}")

    # Store in adata.uns
    if not result_df.empty:
        adata.uns["proportion_test_r_results"] = result_df.to_dict(orient="list")

    return {
        "method": "proportion_test_r",
        "r_success": r_success,
        "n_comparisons": result_df["comparison"].nunique() if not result_df.empty else 0,
        "n_cell_types": result_df["clusters"].nunique() if not result_df.empty else 0,
        "n_significant": int((result_df["FDR"] < 0.05).sum()) if not result_df.empty and "FDR" in result_df.columns else 0,
    }
```
