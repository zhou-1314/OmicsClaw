```r
library(monocle3)
# 1. Build CellDataSet from SCE
cds <- monocle3::new_cell_data_set(
  expression_data = SingleCellExperiment::counts(sce),   # or assay(sce,"X")
  cell_metadata   = as.data.frame(SummarizedExperiment::colData(sce)),
  gene_metadata   = data.frame(gene_short_name=rownames(sce), row.names=rownames(sce))
)
# 2. Inject existing UMAP (avoid re-running UMAP inside monocle3)
SingleCellExperiment::reducedDims(cds)[["UMAP"]] <- as.matrix(
  SingleCellExperiment::reducedDim(sce, "X_umap")   # 2-column matrix
)[, 1:2]
# 3. Cluster (use existing labels if provided via cluster_key)
cds <- monocle3::cluster_cells(cds, cluster_method="louvain", verbose=FALSE)
# 4. Learn principal graph
cds <- monocle3::learn_graph(cds, use_partition=FALSE, verbose=FALSE)
# 5. Order cells (NON-INTERACTIVELY — pass root_pr_nodes directly)
# Auto-select root: the principal node closest to the centroid of the start cluster
root_pr_nodes <- .auto_select_root(cds, cluster_key, root_cluster)
cds <- monocle3::order_cells(cds, root_pr_nodes=root_pr_nodes)
```

Auto-select root node function logic:
```r
.auto_select_root <- function(cds, cluster_key, root_cluster) {
  # Get UMAP coords of all cells in the root cluster
  pr_graph <- monocle3::principal_graph(cds)[["UMAP"]]
  pr_coords <- t(igraph::layout_as_tree(pr_graph))  # WRONG — use monocle3 accessor
  # Correct: get principal graph node positions
  pr_nodes  <- cds@principal_graph_aux[["UMAP"]]$dp_mst
  if (is.null(pr_nodes)) pr_nodes <- t(igraph::layout_with_fr(pr_graph))
  # Find cells in root cluster
  cell_meta <- as.data.frame(SummarizedExperiment::colData(cds))
  if (!is.null(cluster_key) && cluster_key %in% colnames(cell_meta)) {
    root_cells_mask <- as.character(cell_meta[[cluster_key]]) == as.character(root_cluster)
  } else {
    # fallback: use the cluster with most cells
    tbl <- sort(table(cell_meta[[1]]), decreasing=TRUE)
    root_cluster <- names(tbl)[1]
    root_cells_mask <- as.character(cell_meta[[1]]) == root_cluster
  }
  cell_umap <- as.matrix(SingleCellExperiment::reducedDim(cds, "UMAP"))[root_cells_mask, 1:2, drop=FALSE]
  centroid  <- colMeans(cell_umap)
  # Find nearest principal node to centroid
  pr_mat   <- t(as.matrix(pr_nodes))  # n_nodes x 2
  dists    <- sqrt(rowSums((pr_mat - matrix(centroid, nrow=nrow(pr_mat), ncol=2, byrow=TRUE))^2))
  names(dists) <- rownames(pr_mat)
  names(dists)[which.min(dists)]  # return node name e.g. "Y_1"
}
```

```python
def _run_monocle3_r(adata, output_dir, params):
    import warnings, tempfile
    from omicsclaw.core.r_script_runner import RScriptRunner, RScriptError, RScriptTimeoutError

    r_script = R_SCRIPTS_DIR / "sc_monocle3_r.R"
    if not r_script.exists():
        raise FileNotFoundError(f"R script not found: {r_script}")

    # Save adata to a temp h5ad for R (use standard save_h5ad)
    r_work_dir = output_dir / "r_work_monocle3"
    r_work_dir.mkdir(parents=True, exist_ok=True)
    h5ad_path = r_work_dir / "input.h5ad"
    save_h5ad(adata, h5ad_path)

    cluster_key  = params.get("cluster_key", "leiden")
    use_rep      = params.get("use_rep") or "X_umap"
    root_cluster = params.get("root_cluster") or "auto"
    root_pr_node = "auto"  # always auto-select in Python-driven runs

    runner = RScriptRunner(timeout=2400)  # monocle3 learn_graph can be slow
    r_success = True
    try:
        runner.run_script(
            r_script,
            args=[str(h5ad_path), str(r_work_dir), cluster_key, use_rep,
                  root_cluster, root_pr_node],
            expected_outputs=["monocle3_pseudotime.csv"],
            output_dir=r_work_dir,
        )
    except (RScriptError, RScriptTimeoutError) as exc:
        warnings.warn(f"monocle3_r R script failed (Python figures unaffected): {exc}")
        r_success = False

    pt_df = pd.DataFrame()
    traj_df = pd.DataFrame()
    if r_success:
        pt_csv   = r_work_dir / "monocle3_pseudotime.csv"
        traj_csv = r_work_dir / "monocle3_trajectory.csv"
        if pt_csv.exists():
            try:
                pt_df = pd.read_csv(pt_csv)
            except Exception as exc:
                warnings.warn(f"Could not read monocle3 pseudotime CSV: {exc}")
        if traj_csv.exists():
            try:
                traj_df = pd.read_csv(traj_csv)
            except Exception:
                pass

    n_with_pt = 0
    if not pt_df.empty and "monocle3_pseudotime" in pt_df.columns:
        pt_df_indexed = pt_df.set_index("cell_id")
        adata.obs["monocle3_pseudotime"] = pd.to_numeric(
            pt_df_indexed.reindex(adata.obs_names.astype(str))["monocle3_pseudotime"],
            errors="coerce",
        )
        if "monocle3_cluster" in pt_df_indexed.columns:
            adata.obs["monocle3_cluster"] = (
                pt_df_indexed.reindex(adata.obs_names.astype(str))["monocle3_cluster"]
                .astype(str)
            )
        n_with_pt = int(adata.obs["monocle3_pseudotime"].notna().sum())

    if not traj_df.empty:
        adata.uns["monocle3_trajectory"] = traj_df.to_dict(orient="list")

    return {
        "method": "monocle3_r",
        "r_success": r_success,
        "n_cells_with_pseudotime": n_with_pt,
        "cluster_key": cluster_key,
    }
```
