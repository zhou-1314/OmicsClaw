```r
if (!interactive()) {
  args <- commandArgs(trailingOnly=TRUE)
  if (length(args) < 3) {
    cat("Usage: Rscript registry.R <renderer> <data_dir> <out_path> [key=value ...]\n",
        file=stderr())
    quit(status=1)
  }
  renderer_name <- args[1]
  data_dir      <- args[2]
  out_path      <- args[3]
  params        <- parse_kv(args[4:length(args)])

  fn <- R_PLOT_REGISTRY[[renderer_name]]
  if (is.null(fn)) {
    cat("ERROR: Unknown renderer '", renderer_name, "'. Available:",
        paste(names(R_PLOT_REGISTRY), collapse=", "), "\n", sep="", file=stderr())
    quit(status=1)
  }

  tryCatch({
    fn(data_dir, out_path, params)
    cat("SUCCESS:", renderer_name, "->", out_path, "\n")
  }, error=function(e) {
    cat("ERROR in", renderer_name, ":", conditionMessage(e), "\n", file=stderr())
    quit(status=1)
  })
}
```

3. **Implement `call_r_plot(renderer, figure_data_dir, out_path, params=None) -> Path`:**
```python
def call_r_plot(
    renderer: str,
    figure_data_dir: Path,
    out_path: Path,
    params: dict | None = None,
) -> Path:
    """Invoke one R Enhanced renderer via registry.R.

    Parameters
    ----------
    renderer : str
        Name of the renderer function registered in R_PLOT_REGISTRY.
    figure_data_dir : Path
        Directory containing figure_data CSVs (passed as data_dir to R).
    out_path : Path
        Absolute path for the output PNG.
    params : dict, optional
        Extra key=value parameters forwarded to the R function.

    Returns
    -------
    Path
        out_path on success.

    Raises
    ------
    Emits a Python warning on failure — never raises.
    """
    import warnings
    kv_args = [f"{k}={v}" for k, v in (params or {}).items()]
    out_path = Path(out_path)
    out_path.parent.mkdir(parents=True, exist_ok=True)

    try:
        _runner.run_script(
            _REGISTRY_R,
            args=[renderer, str(figure_data_dir), str(out_path)] + kv_args,
            expected_outputs=[out_path.name],
            output_dir=out_path.parent,
        )
    except (RScriptError, RScriptTimeoutError, FileNotFoundError) as exc:
        warnings.warn(
            f"R Enhanced plot '{renderer}' failed (Python figures unaffected): {exc}",
            stacklevel=2,
        )
    return out_path
```
