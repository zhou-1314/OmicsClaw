142: def _handle_auth_command(args) -> None:
143:     """Dispatch ``omicsclaw auth {login,logout,status} [claude|openai]``.
144: 
145:     Thin wrapper over the ``ccproxy`` CLI: we only handle provider-name

147:     multi-provider status view. All OAuth flow logic lives in ccproxy.
148:     """
149:     try:
150:         from omicsclaw.core.ccproxy_manager import (
151:             OAUTH_PROVIDERS,
152:             ccproxy_diagnostic_hint,
153:             ccproxy_executable,
154:             check_ccproxy_auth,
155:             get_oauth_provider,
156:             is_ccproxy_available,
157:             oauth_install_hint,
158:         )
159:     except Exception as exc:
160:         print(f"{RED}Error importing ccproxy_manager: {exc}{RESET}", file=sys.stderr)
161:         sys.exit(2)
162: 
163:     if not is_ccproxy_available():
164:         print(
165:             f"{RED}ccproxy is not installed (from this Python's perspective).{RESET}",
166:             file=sys.stderr,
167:         )
168:         print(ccproxy_diagnostic_hint(), file=sys.stderr)
169:         print(f"Install: {CYAN}{oauth_install_hint()}{RESET}", file=sys.stderr)
170:         sys.exit(2)
171: 
172:     op = getattr(args, "auth_command", None)
173:     target_alias = getattr(args, "provider", None)
174: 
175:     if op is None:
176:         print(
177:             f"Usage: {CYAN}python omicsclaw.py auth [login|logout|status] "
178:             f"[claude|openai]{RESET}"
179:         )
180:         sys.exit(1)
181: 
182:     if op == "status" and not target_alias:
183:         print(f"\n{BOLD}ccproxy OAuth status{RESET}")
184:         print(f"{BOLD}{'=' * 50}{RESET}")
185:         for p in OAUTH_PROVIDERS.values():
186:             ok, msg = check_ccproxy_auth(p.ccproxy_target)
187:             mark = f"{GREEN}OK{RESET}" if ok else f"{YELLOW}--{RESET}"
188:             print(f"  [{mark}] {p.omics_name:<10} {msg}")
189:         sys.exit(0)

191:     if not target_alias:

193:             f"{RED}Error:{RESET} `{op}` requires a provider "
