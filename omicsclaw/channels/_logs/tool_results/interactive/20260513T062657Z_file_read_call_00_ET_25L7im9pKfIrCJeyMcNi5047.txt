109: def _prepare_skill_run(
110:     skill_name: str,
111:     *,
112:     input_path: str | None,
113:     input_paths: list[str] | None,
114:     output_dir: str | None,
115:     demo: bool,
116:     session_path: str | None,
117:     extra_args: list[str] | None,
118:     log_banner: bool = True,
119: ) -> _PreparedSkillRun | SkillRunResult:
120:     """Resolve skill, build argv, prepare output dir. Returns the prepared
121:     run or a stable ``_err`` ``SkillRunResult`` on setup failure.
122: 
123:     ``log_banner`` controls whether the human-readable "Running …" banner
124:     prints; the async executor path keeps it silent because the jobs
125:     router has its own log stream.
126:     """
127:     skills = ensure_registry_loaded().skills
128:     skill_info = skills.get(skill_name)
129:     if skill_info is None:
130:         return _err(skill_name, f"Unknown skill '{skill_name}'. Available: {list(skills.keys())}")
131: 
132:     script_path: Path = skill_info["script"]
133:     if not script_path.exists():
134:         return _err(skill_name, f"Script not found: {script_path}")
135: 
136:     resolved_input_paths: list[str] | None = None
137:     if input_paths and len(input_paths) >= 2:
138:         resolved_input_paths = [str(Path(p).resolve()) for p in input_paths]
139: 
140:     resolved_input = input_path
141:     if session_path and not input_path and not demo and not resolved_input_paths:
142:         from omicsclaw.common.session import SpatialSession
143: 
144:         session = SpatialSession.load(session_path)
145:         if session.h5ad_path:
146:             resolved_input = session.h5ad_path
147: 
148:     if resolved_input:
149:         resolved_input = str(Path(resolved_input).resolve())
150: 
151:     user_supplied_output_dir = output_dir is not None
152:     generated_ts = datetime.now().strftime("%Y%m%d_%H%M%S")
153:     requested_method = extract_flag_value(extra_args, "--method")
154: 
155:     if output_dir:
156:         out_dir = Path(output_dir).resolve()
157:     else:
158:         auto_name = build_output_dir_name(skill_name, generated_ts, method=requested_method)
159:         out_dir = deduplicate_path(DEFAULT_OUTPUT_ROOT / auto_name)
160:     out_dir.mkdir(parents=True, exist_ok=True)
161: 
162:     cmd = build_skill_argv(
163:         python_executable=PYTHON,
164:         script_path=script_path,
165:         skill_info=skill_info,
166:         demo=demo,
167:         input_path=resolved_input,
168:         input_paths=resolved_input_paths,
169:         output_dir=out_dir,
170:     )
171:     if cmd is None:
172:         return _err(skill_name, "No --input, --demo, or --session provided.")
173: 
174:     if log_banner:
175:         domain = skill_info.get("domain", "unknown")
176:         domain_display = registry.domains.get(domain, {}).get("name", domain.title())
177:         if demo:
178:             mode_str = f"{CYAN}demo mode{RESET}"
179:         elif resolved_input_paths:
180:             mode_str = f"inputs: {', '.join(resolved_input_paths)}"
181:         else:
182:             mode_str = f"input: {resolved_input}"
183:         print(f"\n{BOLD}Running {domain_display} skill:{RESET} {GREEN}{skill_name}{RESET} ({mode_str})")
184:         print(f"{BOLD}Output:{RESET} {out_dir}\n")
185: 
186:     filtered = filter_forwarded_args(
187:         extra_args,
188:         allowed_extra_flags=skill_info.get("allowed_extra_flags", set()),
189:     )
190:     cmd.extend(filtered)
191: 
192:     env = os.environ.copy()
193:     env["PYTHONPATH"] = str(OMICSCLAW_DIR) + os.pathsep + env.get("PYTHONPATH", "")
194: 
195:     return _PreparedSkillRun(
196:         skill_name=skill_name,
197:         skill_info=skill_info,
198:         script_path=script_path,
199:         resolved_input=resolved_input,
200:         resolved_input_paths=resolved_input_paths,
201:         out_dir=out_dir,
202:         user_supplied_output_dir=user_supplied_output_dir,
203:         generated_ts=generated_ts,
204:         requested_method=requested_method,
205:         cmd=cmd,
206:         filtered_extra_args=filtered,
207:         env=env,
208:         demo=demo,
209:         session_path=session_path,
210:     )

213: def _finalize_skill_run(
214:     prepared: _PreparedSkillRun,
215:     proc: subprocess.CompletedProcess,
216:     duration: float,
217: ) -> SkillRunResult:
218:     """Run output finalization, build a ``SkillRunResult``, store session.
219: 
220:     Shared between the sync ``run_skill`` and the async ``arun_skill`` so
221:     success / failure shape stays identical regardless of which subprocess
222:     driver fired.
223:     """
224:     final_out_dir = prepared.out_dir
225:     actual_method = prepared.requested_method
226:     readme_path = ""
227:     notebook_path = ""
228:     if proc.returncode == 0:
229:         user_command = build_user_run_command(
230:             skill_name=prepared.skill_name,
231:             demo=prepared.demo,
232:             input_path=prepared.resolved_input,
233:             output_dir=prepared.out_dir,
234:             forwarded_args=prepared.filtered_extra_args,
235:         )
236:         final_out_dir, actual_method, readme_path, notebook_path, _ = finalize_output_directory(
237:             prepared.out_dir,
238:             skill_name=prepared.skill_name,
239:             skill_info=prepared.skill_info,
240:             timestamp=prepared.generated_ts,
241:             user_supplied_output_dir=prepared.user_supplied_output_dir,
242:             preferred_method=prepared.requested_method,
243:             actual_command=user_command,
244:         )
245: 
246:     output_files = sorted(
247:         [path.name for path in final_out_dir.rglob("*") if path.is_file()]
248:     ) if final_out_dir.exists() else []
249: 
250:     result = build_skill_run_result(
251:         skill=prepared.skill_name,
252:         success=proc.returncode == 0,
253:         exit_code=proc.returncode,
254:         output_dir=final_out_dir,
255:         files=output_files,
256:         stdout=proc.stdout,
257:         stderr=proc.stderr,
258:         duration_seconds=duration,
259:         method=actual_method,
260:         readme_path=readme_path,
261:         notebook_path=notebook_path,
262:     )
263: 
264:     if prepared.session_path and result.success:
265:         _store_result_in_session(prepared.session_path, prepared.skill_name, final_out_dir)
266: 
267:     return result

270: def resolve_skill_alias(skill_name: str) -> str:
271:     """Resolve a user-facing skill name or legacy alias to its canonical alias."""
272:     skills = ensure_registry_loaded().skills
273:     if skill_name in skills:
274:         return skills[skill_name].get("alias", skill_name)
275: 
276:     for skill_key, skill_info in skills.items():
277:         legacy_aliases = skill_info.get("legacy_aliases", [])
278:         if skill_name in legacy_aliases:
279:             return skill_key
280: 
281:     if ":" in skill_name:
282:         _domain, skill = skill_name.split(":", 1)
283:         if skill in skills:
284:             return skill
285: 
286:     return skill_name
